-- 회원 테이블
create table tblMember
(
  id varchar2(10) primary key, -- 아이디
  pw varchar2(10) not null, -- 비밀번호
  lv char(1) not null, -- 등급
  image varchar2(100) null, -- 사진
  fileName varchar2(100) null,
  name nvarchar2(4) not null, -- 이름
  birth date not null, -- 생년월일
  age number not null, -- 나이
  gender char(1) not null, -- 성별
  tel varchar2(13) not null, -- 휴대번호
  email varchar2(30) not null -- 이메일
);

-- 일정 테이블
create table tblItinerary
(
  seq number primary key, -- 번호
  id varchar2(10) not null references tblMember(id), -- 아이디
  destination nvarchar2(7) not null, -- 목적지, 여행지  대분류
  startDate date not null, -- 시작날
  endDate date not null, -- 마지막날
  peopleNum number(1) not null, -- 총 인원
  price number not null, -- 가격
  gender char(1) not null -- 성별  회원정보로 자동으로 입력
);
create sequence itinerarySeq;

-- 시특별시 테이블
create table tblCity(
    seq number primary key,
    city nvarchar2(7) not null -- 서울특별시
);
create sequence citySeq;

-- 시지역구역 테이블
create table tblRegion(
    seq number primary key,
    citySeq number not null references tblCity(seq),
    region nvarchar2(10) not null -- 동대문구
);
create sequence regionSeq;

-- 상세일정 테이블
create table tblDetailItinerary
(
  seq number primary key, -- 번호
  itinerarySeq number not null references tblItinerary(seq), -- 일정번호
  detailDate date not null, -- 상세날짜
  time varchar2(10) not null, -- 시간 PM1000
  regionSeq number not null references tblRegion(seq), -- 소분류
  place nvarchar2(30) not null -- 장소
);
create sequence detailItinerarySeq;

-- 숙박 테이블
create table tblAccommodation(
  codeNum varchar2(5) primary key, -- 숙박시설 고유번호 SE001 BU001
  regionSeq number not null references tblRegion(seq),
  name nvarchar2(20) not null, -- 숙박 이름
  address nvarchar2(50) not null,
  latitude number not null, --위도
  longitude number not null, --경도
  mainImg varchar2(100) null,  -- SE001_main.jpg
  fileName varchar2(100) null,
  checkIn varchar2(10) not null, -- AM1100
  checkOut varchar2(10) not null, -- PM0300
  price number not null, -- 가격  88000
  breakFirst char(1) not null, -- 조식 y or n
  totalZzim number not null, -- 총 찜  3
  bathroom number not null, -- 화장실 수  2
  bed number not null, -- 침대 수  3
  room number not null, -- 방 수  1
  gender char(1) not null --성별
);

-- 찜 테이블
create table tblZzim 
(
  seq number primary key, -- 번호
  id varchar2(10) not null references tblMember(id), -- 아이디
  accommodationCodeNum varchar2(5) not null references tblAccommodation(codeNum), -- 숙박코드
  zzimDate date default sysdate null,  -- 찜날짜
  dDay date not null -- 숙박날짜
);
create sequence zzimSeq;

-- 결제 테이블
create table tblPayment 
(
  seq number primary key, -- 번호
  accommodationCodeNum varchar2(5) not null references tblAccommodation(codeNum), -- 숙박코드
  detailItinerarySeq number not null references tblDetailItinerary(seq), -- 상세일정번호
  id varchar2(10) not null references tblMember(id), -- 아이디
  zzimSeq number not null references tblZzim(seq),
  ch char(1) not null -- 체크 - 결제(y) or 취소(n)
);
create sequence paymentSeq;

-- 메시지함 테이블
create table tblMessage 
(
  seq number primary key, -- 번호
  id varchar2(10) not null references tblMember(id), -- 아이디
  title nvarchar2(100) not null,
  message nvarchar2(500) not null, -- 메시지
  regDate date default sysdate not null -- 등록일
);
create sequence messageSeq;

-- 상세이미지 테이블
create table tblDetailImg(
   seq number primary key,
   codeNum varchar2(5) not null references tblAccommodation(codeNum), 
   detailImg1 varchar2(100) null,  -- SE001_detail01.jpg
   fileName1 varchar2(100) null,
   detailImg2 varchar2(100) null,
   fileName2 varchar2(100) null,
   detailImg3 varchar2(100) null,
   fileName3 varchar2(100) null,
   detailImg4 varchar2(100) null,
   fileName4 varchar2(100) null,
   detailImg5 varchar2(100) null,
   fileName5 varchar2(100) null
);
create sequence detailImgSeq;

-- 편의시설 테이블
create table tblAmenity(
   seq number primary key,
   codeNum varchar2(5) not null references tblAccommodation(codeNum),
   internet char(1) not null, -- y or n
   elevator char(1) not null,
   tv char(1) not null,
   washingMachine char(1) not null,
   hairDryer char(1) not null,
   pool char(1) not null,
   parkingLot char(1) not null
);
create sequence amenitySeq;

-- 주변추천 테이블
create table tblAttraction(
    seq number primary key,
    regionSeq number not null references tblRegion(seq),
    image varchar2(20) not null, 
    name nvarchar2(10), -- 해운대 해수욕장
    address nvarchar2(50) not null,
    description nvarchar2(500) not null -- 부산의 자랑 해운대에서 야경을 즐겨보세요.
);
create sequence attractionSeq;

--리뷰 테이블
create table tblReview
(
 seq number primary key,  -- 번호
 id varchar2(10) not null references tblMember(id), --id 회원테이블참조
 codeNum varchar2(5)  not null references tblAccommodation, -- 상품코드넘버 참조
 title nvarchar2(100) not null, -- 글제목
 content nvarchar2(500) not null, -- 글내용
 readCount number default (0) not null, --조회수
 regDate date default sysdate not null -- 등록날짜
);
create sequence reviewSeq;

--숙박Question 테이블
create table tblAccommodationQuestion
(
 seq number primary key, -- 번호
 id varchar2(10) not null references tblMember(id), --id 회원테이블참조
 codeNum varchar2(5) not null references tblAccommodation, -- 상품코드넘버 참조
 title nvarchar2(100) not null, -- 글제목
 content nvarchar2(500) not null,  -- 글내용
 readCount number default (0) not null, --조회수
 regDate date default sysdate not null -- 등록날짜
);
create sequence accommodationQuestionSeq;

--홈페이지Question 테이블
create table tblHomepageQuestion
(
 seq number primary key, -- 번호
 id varchar2(10) not null references tblMember(id), --id 회원테이블참조
 title nvarchar2(100) not null, -- 글제목
 content nvarchar2(500) not null,  -- 글내용
 readCount number default (0) not null, --조회수
 regDate date default sysdate not null -- 등록날짜
); 
create sequence homepageQuestionSeq;

-- 상품 답변 테이블
create table tblAccommodationAnswer
(
 seq number primary key, -- 번호
 id varchar2(10) not null references tblMember(id),--id 회원테이블참조
 answer nvarchar2(500) not null, -- 답변
 accommodationQuestionSeq number not null references tblAccommodationQuestion(seq) -- 상품번호 참조 답변위함
);
create sequence accommodationAnswerSeq;

-- 홈페이지 답변 테이블 
create table tblHomepageAnswer
(
 seq number primary key,
 id varchar2(10) not null references tblMember(id), --id 회원테이블참조
 answer nvarchar2(500) not null, -- 답변
 homepageQuestionSeq number not null references tblHomepageQuestion(seq)  -- 홈페이지번호 참조 답변위함
);
create sequence homepageAnswerSeq; 

-- 공지사항 테이블
create table tblNotice
(
 seq number primary key, -- 번호
 id varchar2(10)  not null references tblMember(id), --id 회원테이블참조
 image varchar2(100) null, -- 사진, notice1.png 여기서 1은 seq야
 fileName varchar2(100) null,
 title nvarchar2(100) not null,
 content nvarchar2(2000) not null, -- 글내용
 readCount number default (0) not null, --조회수
 regDate date default sysdate not null  -- 등록날짜
);
create sequence noticeSeq;
